class Solution {
    public boolean containsDuplicate(int[] nums) {
        HashMap<Integer,Integer> hm=new HashMap<>();
        for(int i=0;i<nums.length;i++)
        {
            int n=nums[i];
            hm.put(n,hm.getOrDefault(n,0)+1);
        }
        for(int key:hm.keySet())
        {
            if(hm.get(key)>1)
            {
                return true;
            }
        }
        return false;
    }
}
