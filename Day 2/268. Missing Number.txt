class Solution {
    public int missingNumber(int[] nums) {
        int n=nums.length;
        int[] arr=new int[n+1];
        for(int i=0;i<=n;i++)
        {
            //int n=arr[i];
            arr[i]=i;
        }
        Arrays.sort(nums);
        Arrays.sort(arr);
        for(int i=0;i<n;i++)
        {
            if(nums[i]!=arr[i])
            {
                return arr[i];
            }
        }
        return arr[n];
    }
}
