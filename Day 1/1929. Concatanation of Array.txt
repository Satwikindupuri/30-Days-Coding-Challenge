Description: Given an integer array nums of length n, you want to create an array ans of length 2n where ans [i] nums [i] and ans [i + n] == nums [i] for 0 i < n (0-indexed). Specifically, ans is the concatenation of two nums arrays. Return the array ans 

Example 1: 
Input: nums = [1,2,1] 
Output: [1,2,1,1,2,1] 
Explanation: The array ans is formed as follows: ans = [nums [0], nums [1], nums [2], nums [0], nums [1], nums [2]] ans = [1,2,1,1,2,1]

ðŸ§¾ Notes: Just A simple approach where just used new array and added the elements twice
*/

class Solution {
    public int[] getConcatenation(int[] nums) {
      int n=2*nums.length;
      int[] arr = new int[n];
      int a=0;

      for(int i=0;i<nums.length;i++)
      {
        arr[a++]=nums[i];
      }  
      for(int i=0;i<nums.length;i++)
      {
        arr[a++]=nums[i];
      }
      return arr;
    }
}
